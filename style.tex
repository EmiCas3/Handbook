% 5. CODE BLOCKS
%    Uses a custom CPP language to avoid listings' broken directive
%    parser which conflicts with C++ template angle-brackets <>.
% ------------------------------------------------------------------
\usepackage{listings}
\usepackage{fvextra}
\usepackage[most]{tcolorbox}

% Custom CPP language: no preprocessor directives, safe with templates
\lstdefinelanguage{CPP}{
  sensitive   = true,
  morecomment = [l]{//},
  morecomment = [s]{/*}{*/},
  morestring  = [b]",
  morestring  = [b]',
  keywords    = {auto,break,case,catch,class,const,constexpr,continue,
                 default,delete,do,else,enum,explicit,extern,false,for,
                 friend,goto,if,inline,int,long,mutable,namespace,new,
                 noexcept,nullptr,operator,private,protected,public,
                 register,return,short,signed,sizeof,static,struct,
                 switch,template,this,throw,true,try,typedef,typename,
                 union,unsigned,using,virtual,void,volatile,while,
                 bool,char,double,float,string,vector,map,set,pair,
                 queue,stack,deque,list,array,bitset,priority_queue,
                 unordered_map,unordered_set,multimap,multiset,
                 ll,vi,vvi,vl,pi,u64,ull,define,include},
}

\lstdefinestyle{cppstyle}{
  language         = CPP,
  basicstyle       = \ttfamily\small,
  keywordstyle     = \bfseries\color{kw},
  stringstyle      = \color{str},
  commentstyle     = \itshape\color{cmt},
  breaklines       = true,
  breakatwhitespace= true,
  tabsize          = 4,
  showstringspaces = false,
  frame            = none,
  aboveskip        = 0pt,
  belowskip        = 0pt,
  xleftmargin      = 0pt,
  % Spanish chars in code comments
  literate =
    {á}{{\'{a}}}1 {é}{{\'{e}}}1 {í}{{\'{i}}}1 {ó}{{\'{o}}}1 {ú}{{\'{u}}}1
    {Á}{{\'{A}}}1 {É}{{\'{E}}}1 {Í}{{\'{I}}}1 {Ó}{{\'{O}}}1 {Ú}{{\'{U}}}1
    {ñ}{{\~{n}}}1 {Ñ}{{\~{N}}}1
    {θ}{{\ensuremath{\theta}}}1,
}

% \codeInput{filename} - renders a .verbatim file in a styled box
\newcommand{\codeInput}[1]{%
  \tcbinputlisting{
    listing only,
    listing options = {style=cppstyle},
    colback  = codebg,
    colframe = codeborder,
    arc      = 3pt,
    boxrule  = 0.5pt,
    left=6pt, right=6pt, top=4pt, bottom=4pt,
    before skip=4pt, after skip=6pt,
    breakable,
    listing file = {#1}
  }%
}

% Inline code pill
\newcommand{\codeinline}[1]{%
  \tcbox[on line,
    colback=codebg, colframe=codeborder,
    arc=2pt, boxrule=0.4pt,
    left=1pt, right=1pt, top=0pt, bottom=0pt,
    boxsep=1pt]{\ttfamily\small #1}}

% ------------------------------------------------------------------
% 6. TABLES
% ------------------------------------------------------------------
\usepackage{colortbl}
\usepackage{array}
\usepackage{etoolbox}

\newcommand{\mdtabularopen}[1]{%
  \begin{center}%
  \rowcolors{2}{rowgray}{white}%
  \setlength{\arrayrulewidth}{0.6pt}%
  \setlength{\tabcolsep}{6pt}%
  \renewcommand{\arraystretch}{1.25}%
  \begin{tabular}{#1}%
}
\newcommand{\mdtabularclose}{\end{tabular}\end{center}}

\newcolumntype{H}{>{\columncolor{headergray}\centering\arraybackslash}c}
\newcolumntype{C}{>{\centering\arraybackslash}c}

\newenvironment{mdtable2}[1][]{\mdtabularopen{|H|C|}}       {\mdtabularclose}
\newenvironment{mdtable3}[1][]{\mdtabularopen{|H|C|C|}}     {\mdtabularclose}
\newenvironment{mdtable4}[1][]{\mdtabularopen{|H|C|C|C|}}   {\mdtabularclose}
\newenvironment{mdtable5}[1][]{\mdtabularopen{|H|C|C|C|C|}} {\mdtabularclose}
