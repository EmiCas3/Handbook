## Sucesiones
\begin{align*}
\text{Números de Fibonacci:} & \quad F(n) = F(n-1) + F(n-2), \quad F(0)=0, \; F(1)=1 \\\\
\text{Números de Lucas:} & \quad L(n) = L(n-1) + L(n-2), \quad L(0)=2, \; L(1)=1 \\\\
\text{Números de Pell:} & \quad P(n) = 2P(n-1) + P(n-2), \quad P(0)=0, \; P(1)=1 \\\\
\text{Números de Mersenne:} & \quad M_n = 2^n - 1, \quad n \text{ primo} \\\\
\text{Números Poligonales:} & \quad P(s, n) = \frac{(s-2)n^2 - (s-4)n}{2}, \quad s\ge 3
\end{align*}

## Sumatorias
\begin{align*}
\text{Suma Gaussiana:} & \quad \sum_{i=1}^{n} i = \frac{n(n+1)}{2} \\\\
\text{Números Triangulares:} & \quad T(n) = \frac{n(n+1)}{2}, \quad T(1)=1 \\\\
\text{Números Cuadrados:} & \quad S(n) = n^2, \quad S(1)=1 \\\\
\text{Números Triangulares Centrales:} & \quad TC(n) = \frac{n(n+1)(2n+1)}{6}, \quad TC(1)=1 \\\\
\text{Números Triangulares Tetraédricos:} & \quad Te(n) = \frac{n(n+1)(n+2)}{6} \\\\
\text{Potencias:} & \quad a^n, \; 2^n, \; 3^n, \dots \\\\
\text{Factoriales:} & \quad n! = 1 \cdot 2 \cdot \dots \cdot n, \quad 0!=1
\end{align*}

## Combinatoria
\begin{align*}
\text{Números de Catalan:} & \quad C(n) = \frac{(2n)!}{(n+1)! \, n!}, \quad C(0)=1 \\\\
\text{Números de Stirling de Segundo Tipo:} & \quad S(n, k) = k \cdot S(n-1, k) + S(n-1, k-1) \\\\
\text{Números de Bell:} & \quad B(n) = \sum_{k=0}^{n} S(n, k)
\end{align*}

## Números Especiales
\begin{align*}
\text{Números Primos:} & \quad \text{no hay fórmula cerrada} \\\\
\text{Números Armónicos:} & \quad H_n = 1 + \frac{1}{2} + \frac{1}{3} + \dots + \frac{1}{n} \\\\\\\
\text{Números Perfectos:} & \quad \sigma(n) = 2n \quad (\text{para números perfectos}) \\\\
\text{Números de Fermat:} & \quad F_n = 2^{2^n} + 1
\end{align*}
